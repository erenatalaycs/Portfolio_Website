    **Step 2.3 — Author `src/scene/colors.test.ts` verbatim:**

    ```ts
    // src/scene/colors.test.ts
    //
    // Drift assertion: SCENE_COLORS must match the corresponding @theme tokens
    // in src/styles/tokens.css. Editing one file without updating the other
    // fails this test, gating CI.
    //
    // Source: 02-RESEARCH.md Pattern 8 § "Drift test (single-file, jsdom-free)"

    import { describe, it, expect } from 'vitest';
    import { readFileSync } from 'node:fs';
    import { fileURLToPath } from 'node:url';
    import { dirname, resolve } from 'node:path';
    import { SCENE_COLORS } from './colors';

    const __dirname = dirname(fileURLToPath(import.meta.url));

    describe('SCENE_COLORS mirrors tokens.css', () => {
      const tokensCss = readFileSync(resolve(__dirname, '../styles/tokens.css'), 'utf-8');

      function token(name: string): string {
        const re = new RegExp(`--color-${name}:\\s*(#[0-9a-fA-F]{6})`);
        const m = tokensCss.match(re);
        if (!m) throw new Error(`tokens.css missing --color-${name}`);
        return m[1].toLowerCase();
      }

      it.each([
        ['bg', 'bg'],
        ['surface', 'surface-1'],
        ['accent', 'accent'],
        ['warn', 'warn'],
      ])('SCENE_COLORS.%s matches --color-%s', (sceneKey, tokenName) => {
        expect(SCENE_COLORS[sceneKey as keyof typeof SCENE_COLORS].toLowerCase()).toBe(
          token(tokenName),
        );
      });
    });
    ```

    Notes on this exact shape:
    - `readFileSync(__dirname + '/../styles/tokens.css')` — Vitest runs from the repo root, but `__dirname` is constructed from `import.meta.url` (ESM-safe) so the path is anchored to this test's location. This works regardless of where Vitest is invoked from.
    - `it.each([...])` — Vitest table-driven test runs four assertions (one per mirrored key), each with its own pass/fail line in the reporter. If `SCENE_COLORS.surface` drifts from `tokens.css`'s `--color-surface-1`, ONLY that row fails — the others still report.
    - The regex `--color-${name}:\s*(#[0-9a-fA-F]{6})` matches the exact format used in `tokens.css` (a colon, optional whitespace, then a 6-char hex value). Tested mentally against the lines `--color-bg: #0d1117;` and `--color-surface-1: #161b22;` — both match. The regex does NOT support 3-char hex (`#fff`) or 8-char hex (`#0d1117ff`) — those formats are not used in `tokens.css` per Phase 1's lock.
    - The `.toLowerCase()` call on both sides normalizes — `#0D1117` and `#0d1117` are equivalent in CSS but would compare unequal as strings.
