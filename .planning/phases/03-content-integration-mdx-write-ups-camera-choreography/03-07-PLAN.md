    #!/usr/bin/env node
    // scripts/check-parity.mjs
    //
    // TXT-06 + CNT-03 + Pitfall 8 audit. Runs in CI BEFORE `vite build`
    // (gated via package.json scripts.build chain). Exits non-zero with a
    // human-readable error message on drift; exits 0 on pass.
    //
    // Three assertions:
    //  1. TXT-06 — every SECTIONS id has a text-shell render OR a 3D mount
    //  2. CNT-03 — every skills.ts tag appears in some project tagline OR
    //              some write-up frontmatter `tags`
    //  3. Pitfall 8 — every ATT&CK id in writeup frontmatter exists as a
    //                 key in src/content/attack-techniques.ts
    //
    // Hand-maintained 3D mount mapping (RESEARCH Pattern 11) — JSX AST
    // parsing is fragile; the const below is THE single source of truth for
    // which sections are covered by 3D-shell monitor mounts.
    //
    // Source: 03-RESEARCH.md Pattern 11; 03-UI-SPEC.md § Section parity audit;
    //         03-CONTEXT.md D-19

    import { readFileSync, readdirSync, existsSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __dirname = dirname(fileURLToPath(import.meta.url));
    const ROOT = resolve(__dirname, '..');

    // Hand-maintained mapping. UPDATE if Phase 3+ adds another monitor mount.
    const THREE_D_MOUNTS = {
      left: ['projects'],
      center: ['about', 'skills', 'whoami'],
      right: ['writeups'],
    };

    const errors = [];

    // --- Assertion 1: TXT-06 section parity ---
    const sectionsSrc = readFileSync(resolve(ROOT, 'src/content/sections.ts'), 'utf8');
    const sectionIds = [...sectionsSrc.matchAll(/id:\s*'([^']+)'/g)].map((m) => m[1]);

    const sectionFiles = readdirSync(resolve(ROOT, 'src/sections'))
      .filter((f) => f.endsWith('.tsx'))
      .map((f) => f.replace(/\.tsx$/, '').toLowerCase());

    const threeDSatisfies = new Set(Object.values(THREE_D_MOUNTS).flat());

    for (const id of sectionIds) {
      const hasTextShell = sectionFiles.includes(id);
      const has3D = threeDSatisfies.has(id);
      if (!hasTextShell && !has3D) {
        errors.push(
          `TXT-06: Section "${id}" has neither a text-shell render (src/sections/${id}.tsx) nor a 3D-shell monitor mount (THREE_D_MOUNTS in scripts/check-parity.mjs).`,
        );
      }
    }

    // --- Assertion 2: CNT-03 skills provenance ---
    const skillsSrc = readFileSync(resolve(ROOT, 'src/content/skills.ts'), 'utf8');
    // Extract quoted lowercase tokens that look like skill names. Filter out
    // common non-skill quoted tokens (TypeScript reserved words, comments).
    // The regex is permissive but the parity check is permissive too: false
    // positives only mean "this skill was found in a project/writeup", which
    // is not an audit failure.
    const skillsRaw = [...skillsSrc.matchAll(/['"`]([a-z0-9+#.&/-]+)['"`]/gi)]
      .map((m) => m[1].toLowerCase())
      // Filter out obvious non-skill tokens (TS keywords, type names).
      .filter((t) => !['skilltag', 'name', 'string'].includes(t));

      ? readdirSync(writeupsDir)
          .filter((f) => f.endsWith('.mdx'))
          .flatMap((f) => {
            const src = readFileSync(resolve(writeupsDir, f), 'utf8');
            const tagsBlock = src.match(/tags:\s*\[([^\]]+)\]/);
            if (!tagsBlock) return [];
            return [...tagsBlock[1].matchAll(/['"`]?([a-z0-9+#.&/-]+)['"`]?/gi)].map((m) =>
              m[1].toLowerCase(),
            );
          })
      : [];

      if (!skill || skill.length < 2) continue; // skip noise
      const inProject = projectsSrcLower.includes(skill);
      const inWriteup = writeupTags.includes(skill);
      if (!inProject && !inWriteup) {
        errors.push(
          `CNT-03: skill "${skill}" has no project or write-up demonstrating it. Either add a project mentioning it (projects.ts tagline) or a write-up with it in frontmatter tags, OR remove the skill from skills.ts.`,
        );
      }
    }

      [...attackSrc.matchAll(/['"`](T\d{4}(?:\.\d{3})?)['"`]/g)].map((m) => m[1]),
    );

    if (existsSync(writeupsDir)) {
      for (const f of readdirSync(writeupsDir).filter((f) => f.endsWith('.mdx'))) {
        const src = readFileSync(resolve(writeupsDir, f), 'utf8');
        const block = src.match(/attack_techniques:\s*\n?((?:\s*-\s*[^\n]+\n)+)/);
        if (!block) continue;
        const ids = [...block[1].matchAll(/-\s*(T\d{4}(?:\.\d{3})?)/g)].map((m) => m[1]);
        for (const id of ids) {
          if (!attackKeys.has(id)) {
            errors.push(
              `Pitfall 8: ATT&CK id "${id}" referenced in ${f} but missing from src/content/attack-techniques.ts. Add the canonical name from https://attack.mitre.org/techniques/${id.replace('.', '/')}/ to the lookup table.`,
            );
          }
        }
      }
    }

    // --- Verdict ---
    if (errors.length > 0) {
      console.error(`check-parity.mjs FAILED — ${errors.length} violation(s):`);
      for (const e of errors) console.error('  ' + e);
      process.exit(1);
    }
    console.log(
      `check-parity.mjs PASSED — ${sectionIds.length} sections audited, ${skillsRaw.length} skills audited, ${attackKeys.size} ATT&CK keys checked.`,
    );
